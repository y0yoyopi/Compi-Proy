int main() {
    int a, b, c, d, e, acc;
    a = 2;
    b = 3;
    c = 5;
    d = 7;
    e = 11;
    acc = 0;

    int i;
    i = 0;
    while (i < 2000000) {
        acc = acc + (a + b) + (c + d) + e - (a * b) + (c * d) - (a + b + c + d + e);
        i = i + 1;
    }

    print("-- Resultado variables --");
    print(acc);
    return 0;
}
